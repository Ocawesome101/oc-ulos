NAME
  acls - some ACL-related data

DESCRIPTION
  acls is a table containing permission data for users and files.

  {
    user = {
      SUDO = number,
      MOUNT = number,
      OPEN_UNOWNED = number,
      COMPONENTS = number,
      HWINFO = number,
      SETARCH = number,
      MANAGE_USERS = number,
      BUUTADDR = number,
    },
    file = {
      OWNER_READ = number,
      OWNER_WRITE = number,
      OWNER_EXEC = number,
      GROUP_READ = number,
      GROUP_WRITE = number,
      GROUP_EXEC = number,
      OTHER_READ = number,
      OTHER_WRITE = number,
      OTHER_EXEC = number,
    }
  }
